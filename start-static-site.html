git clone {git-repo-url}

source venv/bin/activate
pip install pelican markdown
pelican-quickstart

mkdir -p templates/mytheme/{static,templates}
touch templates/mytheme/templates/base.html
touch templates/mytheme/static/style.css

Add below into templates/mytheme/templates/base.html.

{% extends "!simple/base.html" %}

{% block head %}
{{ super() }}

Add below into emplates/mytheme/static/style.css

html, body {
margin: 0;
border: 0;
font-size: 100%;
background-color: #000000;
color: #ffffff;
}

Add below theme to your pelicanconf.py

THEME = 'templates/mytheme'

In publishconf.py, Comment out CATEGORY_FEED_ATOM
